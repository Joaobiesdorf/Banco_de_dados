CREATE DATABASE companhia_aerea;

\c companhia_aerea

CREATE TABLE aeroporto (
    id_aeroporto SERIAL PRIMARY KEY,
    nome VARCHAR(100),
    cidade VARCHAR(100),
    sigla CHAR(3)
);

CREATE TABLE voo (
    id_voo SERIAL PRIMARY KEY,
    codigo_voo VARCHAR(10),

    origem_id INT REFERENCES aeroporto(id_aeroporto),
    destino_id INT REFERENCES aeroporto(id_aeroporto),

    horario_saida_previsto TIMESTAMP,
    horario_chegada_previsto TIMESTAMP,

    status VARCHAR(20)
);

CREATE TABLE posicao_voo (
    id_posicao SERIAL PRIMARY KEY,

    voo_id INT REFERENCES voo(id_voo),

    latitude DECIMAL(9,6),
    longitude DECIMAL(9,6),

    percentual_trajeto INT,

    previsao_chegada_atualizada TIMESTAMP
);
